Example 4:

function Get-DisconnectedMailbox {
  param(
  [String]$Name = '*',
  [Switch]$Archive
  )
  
  $databases = Get-MailboxDatabase
  $databases  | %{
    $db = Get-Mailboxstatistics -Database $_ | 
	  ?{$_.DisconnectDate -and $_.IsArchiveMailbox -eq $Archive}
	
    $db | ?{$_.displayname -like $Name} |
      Select DisplayName,
        MailboxGuid,
        Database,
        DisconnectReason
    }
}
